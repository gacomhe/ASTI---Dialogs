Basic dialog prompt

$.astiAlert({
   title: "Success Prompt",	
   content: ["This is a sample prompt message"],
   type: "success",    
});

4 type that can be used : "success", "warning", "error", "cancel"

* Note that you can have two or more contents inside the dialog.["Sample Data 1", 
"Sample Data 2"].
* Note 1st content is the only content with left-mark icon
Success Warning Error Cancel

Auto Close

$.astiAlert({
   title: "Success Prompt",	
   content: ["This is a sample prompt message"],
   type: "success",
   autoClose: true,
   autoCloseTimer: 4000,
});

Dialog with Header Closer

$.astiAlert({
   title: "Success Prompt",	
   content: ["This is a sample prompt message"],
   type: "success",
   headerCloseButton: true, 
});

Dialog with Ok and Cancel Button

$.astiAlert({
   title: "Success Prompt",	
   content: ["This is a sample prompt message"],
   type: "success",
   headerCloseButton: true,
   showOkButton: true,
   showCancelButton: true,    
});

Handling Buttons Action

$.astiAlert({
   title: "Success Prompt",	
   content: ["This is a sample prompt message"],
   type: "success",
   headerCloseButton: true,
   showOkButton: true,
   showCancelButton: true,  
   okButtonAction: function(){
   	alert("Note that if you handle Ok button action the default 
   	closing of dialog will be alter or remove");

   	$.astiAlert("hide");
   },
   cancelButtonAction: function(){
   	alert("Note that if you handle Cancel button action the default 
   	closing of dialog will be alter or remove");
   	
   	$.astiAlert("hide");
   },
   headerButtonAction: function(){
   	alert("Note that if you handle Cancel button action the default 
   	closing of dialog will be alter or remove");
   	
   	$.astiAlert("hide");
   },
});

Custom Hiding

$.astiAlert({
   title: "Success Prompt",	
   content: ["This is a sample prompt message"],
   type: "success",
   headerCloseButton: true,
   showOkButton: true,
   showCancelButton: true,
   animation: "fadeIn",
   okButtonAction: function(){
   	$.astiAlert("hide", "bounce");
   },
   cancelButtonAction: function(){
   	$.astiAlert("hide", "bounce");
   },
   headerButtonAction: function(){
   	$.astiAlert("hide", "bounce");
   },
});
